# Docker_test1

# Install Docker ( for Ubuntu)

    sudo apt-get update
    sudo apt install docker.io
    sudo systemctl start docker
    sudo systemctl enable docker

# Usage

- Clone the repo
- Open terminal into it

# Open Android Studio

    ./run.sh a
  
# Open Eclipse (Java)

    ./run.sh j
    
# Open bash

    ./run.sh /bin/bash
